# Load data
try:
    data = pd.read_csv('/content/sample_data/predictive_maintenance.csv', sep=',')
except FileNotFoundError:
    print("Error: 'predictive_maintenance.csv' not found.")
    exit()
In[3]:
# Data preprocessing
# Drop irrelevant columns
data = data.drop(['UDI', 'Product ID', 'Failure Type'], axis=1)
In[4]:
# Convert categorical 'Type' to numeric using one-hot encoding
data = pd.get_dummies(data, columns=['Type'], prefix='Type')
In[5]:
# Drop rows with NaN
data = data.dropna()
In[6]:
# Features (X) and target (y)
X = data.drop('Target', axis=1)
y = data['Target']
In[7]:
# Normalize the data
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)
In[8]:
# Split the data
X_train, X_test, y_train, y_test = train_test_split(X_scaled, y, test_size=0.2, random_state=42, stratify=y)
In[9]:
# Prepare data for LSTM and CNN (reshape to 3D: [samples, timesteps, features])
timesteps = 1  # Treat each sample as a single timestep
X_train_lstm = X_train.reshape((X_train.shape[0], timesteps, X_train.shape[1]))
X_test_lstm = X_test.reshape((X_test.shape[0], timesteps, X_test.shape[1]))
In[10]:
# Convert y to categorical for LSTM and CNN
y_train_cat = to_categorical(y_train)
y_test_cat = to_categorical(y_test)
In[11]:
# Initialize results list
results = []

# Function to plot confusion matrix
def plot_confusion_matrix(y_true, y_pred, title, filename):
    cm = confusion_matrix(y_true, y_pred)
    plt.figure(figsize=(6, 4))
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues')
    plt.title(f'Confusion Matrix - {title}')
    plt.ylabel('True Label')
    plt.xlabel('Predicted Label')
    plt.savefig(filename)
    print(f"Saved plot: {filename}")
    plt.show()  # Display the plot
    files.download(filename)  # Download the file in Colab
    plt.close()

# 1. Random Forest
print("Training Random Forest...")
start_time = time.time()
rf_model = GridSearchCV(
    RandomForestClassifier(random_state=42, class_weight='balanced'),
    {'n_estimators': [100, 200], 'max_depth': [10, 20]},
    cv=3, scoring='accuracy'
)
rf_model.fit(X_train, y_train)
rf_time = time.time() - start_time
rf_pred = rf_model.predict(X_test)
rf_accuracy = accuracy_score(y_test, rf_pred)
results.append(['Random Forest', rf_accuracy, rf_time])
print("Random Forest Classification Report:")
print(classification_report(y_test, rf_pred))

# 4. LSTM
print("Training LSTM...")
start_time = time.time()
lstm_model = Sequential([
    LSTM(64, input_shape=(timesteps, X_train.shape[1]), return_sequences=False),
    Dense(32, activation='relu'),
    Dropout(0.2),
    Dense(2, activation='softmax')
])
lstm_model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])
lstm_model.fit(X_train_lstm, y_train_cat, epochs=10, batch_size=32, validation_split=0.2, verbose=0)
lstm_time = time.time() - start_time
lstm_pred = lstm_model.predict(X_test_lstm, verbose=0)
lstm_pred = np.argmax(lstm_pred, axis=1)
lstm_accuracy = accuracy_score(y_test, lstm_pred)
results.append(['LSTM', lstm_accuracy, lstm_time])
print("LSTM Classification Report:")
print(classification_report(y_test, lstm_pred))
plot_confusion_matrix(y_test, lstm_pred, 'LSTM', '/content/lstm_confusion_matrix.png')
